Modul: Events

Kann jedem beliebigen Object die Moeglichkeit auf Events zu reagieren und auszuloesen 


var object = {};
_.extend(object, Backbone.Events);
object.on("alert", function(msg) {
  alert("Triggered " + msg);
});
object.trigger("alert", "an event");

Modul: Models

Boilerplate fuer Models. Bietet Eventssupport, getter/setter, Validierung, XHR Schnittstelle 


var Meal = Backbone.Model.extend({
  defaults: {
    "appetizer":  "caesar salad",
    "entree":     "ravioli",
    "dessert":    "cheesecake"
  }
});
var dinner = new Meal();
dinner.on("change:entree", function(...){...});
dinner.set("entree", "maccaroni");
alert("Dessert will be " + dinner.get('dessert'));
